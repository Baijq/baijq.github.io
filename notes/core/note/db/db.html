# SQL部分

如图，我们想实现这个数据查询，怎么做？

解决思路，CASE WHEN 实现

SELECT * FROM tb_stu_cause;

SELECT 姓名,
CASE WHEN 课程 = '语文' THEN 分数 END AS '语文',
CASE WHEN 课程 = '数学'	THEN 分数 END AS '数学'
FROM tb_stu_cause;

SELECT 姓名,
MAX(CASE WHEN 课程 = '语文' THEN 分数 END) AS '语文',
MAX(CASE WHEN 课程 = '数学' THEN 分数 END) AS '数学'
FROM tb_stu_cause
GROUP BY 姓名;
