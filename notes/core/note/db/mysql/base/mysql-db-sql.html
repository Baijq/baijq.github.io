# 二、增删改查

DROP DATABASE IF EXISTS admin;
-- 创建数据库 admin
CREATE DATABASE admin;

-- 切换到 admin
USE admin;

-- 创建表 t_user
DROP TABLE IF EXISTS t_user;
CREATE TABLE t_user(
	id INT PRIMARY KEY AUTO_INCREMENT,
	user_name VARCHAR(10) NOT NULL,
	create_time TIMESTAMP NULL
);
-- 创建表 t_member
CREATE TABLE IF NOT EXISTS t_member(
    id INT UNSIGNED AUTO_INCREMENT COMMENT '主键ID',
    name VARCHAR(8) NOT NULL COMMENT '姓名',
    address VARCHAR(255) COMMENT '地址',
    create_date DATE COMMENT '创建时间',
    PRIMARY KEY(id)
)ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 初始化数据
INSERT INTO admin.t_user (id, user_name, create_time)
VALUES (DEFAULT, '张飞', NOW());
INSERT INTO admin.t_user (id, user_name, create_time)
VALUES (DEFAULT, '孙尚香', NOW());

-- 新增列
ALTER TABLE admin.t_user ADD update_time TIMESTAMP NULL;

-- 更新数据
UPDATE admin.t_user SET update_time = NOW() WHERE id = 1;

-- 删除某行
DELETE FROM admin.t_user WHERE id = 2;

-- SQL SECURITY DEFINER
