  1. 合并数据
merge into productdb.tb_product_group g
using (select u.companyid, u.id from onlinedb.tb_auth_user u) u1
on (g.createuser = u1.id)
when matched then update set g.companyid = u1.companyid


merge into productdb.tb_product_group g
